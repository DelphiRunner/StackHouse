 FUNC LOOP LOCAL:2 STACK:4
  MOV 1 .L0 
  MOV 0 .L1 
  PUSH .L0 100000 
  JN INT <= 9 
  PUSH .L1 .L0 
  IADD 
  POP .L1 
  INC .L0 1 
  JUMP 2 
  PUSH .L1 0 
  CALL %PRINTL 
  RET 
 END FUNC
